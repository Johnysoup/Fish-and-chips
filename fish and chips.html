        let currentStep = 'start';


        function updateGame(step) {

            const state = gameData[step];

            document.getElementById('game-text').textContent = state.text;

            document.getElementById('message').textContent = '';

            document.getElementById('feedback').textContent = '';

            document.getElementById('feedback').style.backgroundColor = '';

            

            const choicesContainer = document.getElementById('choices');

            choicesContainer.innerHTML = '';

            state.choices.forEach(choice => {

                const button = document.createElement('button');

                button.textContent = choice.text;

                button.className = 'button';

                button.onclick = () => makeChoice(choice);

                choicesContainer.appendChild(button);

            });

        }


        function makeChoice(choice) {

            if (choice.message) {

                document.getElementById('message').textContent = choice.message;

            }

            

            if (choice.feedback) {

                const feedbackElem = document.getElementById('feedback');

                feedbackElem.textContent = choice.feedback;

                if (choice.correct) {

                    feedbackElem.style.color = 'white';

                    feedbackElem.style.backgroundColor = 'green';

                } else {

                    feedbackElem.style.color = 'white';

                    feedbackElem.style.backgroundColor = 'red';

                }

            }

            

            currentStep = choice.nextStep;

            updateGame(currentStep);

        }
